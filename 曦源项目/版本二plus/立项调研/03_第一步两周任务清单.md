```python
class FSFProjector(nn.Module):
    def __init__(self, student_dim=4096, hidden_dim=2048, teacher_dim=1024, n_patch=256):
        super().__init__()
        self.bn = nn.BatchNorm1d(student_dim)
        self.mlp = nn.Sequential(
            nn.Linear(student_dim, hidden_dim),
            nn.GELU(),
            nn.Linear(hidden_dim, teacher_dim),
        )
        self.register_buffer('pos_enc', build_2d_sinusoidal_pe(n_patch, teacher_dim))
    
    def forward(self, x_student):
        B, N, _ = x_student.shape
        x = self.bn(x_student.reshape(B*N, -1)).reshape(B, N, -1)
        x = self.mlp(x)
        x = x + self.pos_enc[None]
        return x
```

**关键产出**：unit test 通过，forward (1, 256, 4096) → (1, 256, 1024)

## 附录 A · 方向词 GT 标注脚本（继承 v2）

```python
def annotate_directions(demo_trajectory):
    """
    输入：LIBERO demo (hdf5)，含 robot0_eef_pos 时序与 object_pos 时序
    输出：每步的方向词标签 ∈ {right, left, up, down, front, back, grasped}
    """
    direction_labels = []
    EPS_GRASP = 0.03  # 抓取距离阈值 (m)

    for t in range(demo_trajectory.length):
        eef_pos = demo_trajectory['observations/robot0_eef_pos'][t]
        object_pos = demo_trajectory['observations/object_pos'][t]
        gripper_closed = demo_trajectory['observations/gripper'][t] < 0.05

        # 1. 检查 grasped 状态
        dist = np.linalg.norm(eef_pos - object_pos)
        if dist < EPS_GRASP and gripper_closed:
            direction_labels.append('grasped')
            continue

        # 2. 计算方向向量并取主轴
        dir_vec = object_pos - eef_pos
        abs_dir = np.abs(dir_vec)
        main_axis = np.argmax(abs_dir)

        # 3. 映射到 6 方向
        if main_axis == 0:
            label = 'right' if dir_vec[0] > 0 else 'left'
        elif main_axis == 1:
            label = 'front' if dir_vec[1] > 0 else 'back'
        else:
            label = 'up' if dir_vec[2] > 0 else 'down'

        direction_labels.append(label)

    # 4. 滑动平均 3 步去抖
    smoothed = sliding_majority_vote(direction_labels, window=3)
    return smoothed
```

## 附录 B · Focus Mask 3 源融合伪代码

```python
def make_focus_mask(image, p_eef, p_obj, K, R, t,
                    sam2, grounding_dino, dinov2,
                    target_text, n_patch=256):
    """
    输入：image, robot/object 3D pose, 相机参数, 模型, 物体文本
    输出：(256,) focus mask ∈ [0.1, 1.0]^256（三段量化）
    """
    # Source A: 方向词 GT 倒推 2D Gaussian
    m_A = focus_mask_source_A(p_eef, p_obj, K, R, t)
    
    # Source B: SAM2 物体硬 mask
    m_B = focus_mask_source_B(image, target_text, sam2, grounding_dino)
    
    # Source C: DINOv2 CLS attention
    m_C = focus_mask_source_C(image, dinov2)
    
    # F1 融合 + 三段量化
    m_raw = 0.3 * m_A + 0.4 * m_B + 0.3 * m_C
    m = np.where(m_raw >= 0.7, 1.0,
        np.where(m_raw >= 0.3, 0.5, 0.1))
    return m
```
